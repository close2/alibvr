16 /*
17 Code walk through with an example:
18 11 03 00 6B 00 01
19 [0] 11 → addressId
20 [1] 03 → ReadHoldingRegisters
21 [2][3] 00 6b → 107 (modbus spec name is 40108 because for holding registers the
22  naming convention is to add 40001 to the "real" address)
23 [4][5] 00 01 → number of registers to read (we only support 1; see below)
24 which becomes:
25 : 1 1 0 3 0 0 6 B 0 0 0 1 8 0 CR LF
26  [0] [1] [2] [3] [4] [5] [6]
27 SerialModbus starts with state = Receiving_Waiting_For_Start and crc_c = 0 and partial_char 0xFF;
28 rx(c) will discard everything until the first ':'
29  rx(':') state ← Receiving_Address_Id
30 [0h]rx('1') partial_char ← 1; state stays
31 [0l]rx('1') last_byte ← 11; state ← Receiving_Data; crc ← 11; partial_char ← 0xFF
32 [1h]rx('0') partial_char ← 0; state stays
33 [1l]rx('1') Modbus::rx(last_byte/11); last_byte ← 01; state stays; crc ← 12; partial_char ← 0xFF
34 [2h]rx('0') partial_char ← 0; state stays
35 [2l]rx('0') Modbus::rx(last_byte/01); last_byte ← 01; state stays; crc ← 12; partial_char ← 0xFF
36 [3h]rx('6') partial_char ← 6; state stays
37 [3l]rx('B') Modbus::rx(last_byte/00); last_byte ← 01; state stays; crc ← 7d; partial_char ← 0xFF
38 [4h]rx('0') partial_char ← 0; state stays
39 [4l]rx('0') Modbus::rx(last_byte/6b); last_byte ← 01; state stays; crc ← 7d; partial_char ← 0xFF
40 ...
41 [6h]rx('8') partial_char ← 8; state stays
42 [6l]rx('0') Modbus::rx(last_byte/01); last_byte ← 80; state stays; crc ← 00; partial_char ← 0xFF
43  rx(CR) verify crc == 0; last_byte ← CR; everything else stays
44  rx(LF) verify last_byte = CR; state ← Sending_Start; crc ← 0; partial_char ← 0xFF; Modbus::process
45 
46 Modbus therefore receives:
47 reset; rx: 11, 03, 00, 6B, 00, 01; followed by process
48 
49 If Modbus wants to send a response it will no longer accept character → Modbus::can_rx() returns false;
50 Modbus should not do long processing inside the process() function, because we are still inside
51 an irq handler!
52 When Modbus has finished building the response it will initiate the transfer by
53 calling SerialModbus::tx_start (Notice that SerialModbus is a template argument
54 of Modbus. SerialModbus will then enable irqs for Uart transmitting.)
55 
56 For irq sending Uart calls SerialModbus::tx_is_empty() and tx_pop()
57 SerialModbus is a template argument of Uart!
58 During transmission partial_char is used for storing 4bit-values (same as during
59 reception) and state is used to as flag to know if crc, cr and lf have already be
60 sent.
61 
62 SerialModbus simply wraps modubs tx, rx,... and splits/combines bytes to 4bit
63 representation and adds CRC, CR, LF.
64 */

154  static inline void rx(uint8_t c) {
155  _serial_modbus::State& state = get_state();
156  if ((uint8_t)state > 0x0F /* Sending */ || !_Modbus::can_rx()) {
157  // response has probably not yet been build
158  // ignore all characters until previous message has been processed
159  return;
160  }
161  if (c == _serial_modbus::Start) {
162  reset();
163  state = _serial_modbus::Receiving_Address_Id;
164  return;
165  }
166  if (state == _serial_modbus::Receiving_Waiting_For_Start) {
167  // waiting for _serial_modbus::Start
168  return;
169  }
170 
171  uint8_t& last_byte = get_last_byte();
172  uint8_t& partial_char = get_partial_char();
173 
174  uint8_t& crc = get_crc_c();
175  if (c == _serial_modbus::End1) {
176  // don't send last_byte to Modbus
177  if (crc != 0) {
178  reset();
179  return;
180  }
181  last_byte = _serial_modbus::End1;
182  return;
183  } else if (c == _serial_modbus::End2) {
184  if (last_byte != _serial_modbus::End1) {
185  reset();
186  return;
187  }
188  // prepare for reply
189  crc = 0;
190  partial_char = 0xFF;
191  state = _serial_modbus::Sending_Start;
192  // and tell Modbus to process the command
193  _Modbus::process();
194  return;
195  } else if (last_byte == _serial_modbus::End1 || last_byte == _serial_modbus::End2) {
196  reset();
197  return;
198  }
199 
200  // For every byte after the start-byte we build the value in half-bytes
201  if (partial_char == 0xFF) {
202  partial_char = Adapter::hex_to_v4bit(c);
203  if (partial_char > 0x0F) {
204  reset();
205  }
206  return;
207  }
208 
209  // we have received the second 4bits of a byte.
210  // send Modbus the last completely received byte (making last_byte available again).
211  // unless this is the first character we have received
212  if (state == _serial_modbus::Receiving_Data) {
213  _Modbus::rx(last_byte);
214  }
215 
216  // store byte in last_byte
217  last_byte = Adapter::hex_to_v4bit(c); // first store first half
218  if (last_byte > 0x0F) {
219  reset();
220  return;
221  }
222  last_byte |= partial_char << 4; // then add second half
223 
224  partial_char = 0xFF;
225  add_crc(last_byte);
226 
227  // shortcut to stop receiving if it is another addressId:
228  if (state == _serial_modbus::Receiving_Address_Id) {
229  if (last_byte != 0 && last_byte != addressId) {
230  reset();
231  return;
232  }
233  state = _serial_modbus::Receiving_Data;
234  }
235  }

241  const uint8_t empty = _Modbus::tx_is_empty();
242  _serial_modbus::State& state = get_state();
243 
244  if (state == _serial_modbus::Sending_Start && empty) {
245  // Nothing to send, prepare for Receiving
246  reset();
247  return 1;
248  }
249  // Sending_Start → before sending :
250  // Sending_Data → before or during sending data from Modbus
251  // and sending crc
252  // Sending_End1 → before sending End1 (CR)
253  // Sending_End2 → before sending End2 (LF)
254  // Sending_Done → Done
255  if (state == _serial_modbus::Sending_Done) {
256  // finished Sending, prepare for Receiving
257  reset();
258  return 1;
259  }
260  return 0;
261  }

263  static uint8_t tx_pop() {
264  _serial_modbus::State& state = get_state();
265  if (state == _serial_modbus::Sending_Start) {
266  state = _serial_modbus::Sending_Data;
267  return ':';
268  }
269 
270  uint8_t& partial_char = get_partial_char();
271  if (partial_char != 0xFF) {
272  uint8_t ret = partial_char;
273  partial_char = 0xFF;
274  return ret;
275  }
276 
277  switch (state) {
278  case _serial_modbus::Sending_End1:
279  state = _serial_modbus::Sending_End2;
280  return _serial_modbus::End1;
281  case _serial_modbus::Sending_End2:
282  state = _serial_modbus::Sending_Done;
283  return _serial_modbus::End2;
284  default:
285  if (_Modbus::tx_is_empty()) {
286  state = _serial_modbus::Sending_End1; // switch to Sending_End1 (CR)
287  const uint8_t& crc = get_crc_c(); // but send CRC first.
288  return _split_into_4bits(-crc);
289  }
290  const uint8_t& c = _Modbus::tx_pop();
291  add_crc(c);
292  return _split_into_4bits(c);
293  }
