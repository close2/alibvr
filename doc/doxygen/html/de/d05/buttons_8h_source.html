109  static inline void handle() {
110  typedef typename Button::Pin Pin;
111  if (Pin::port == _macros::B && I::irq != _irqs::IRQ_PCINT0) return;
112  if (Pin::port == _macros::C && I::irq != _irqs::IRQ_PCINT1) return;
113  if (Pin::port == _macros::D && I::irq != _irqs::IRQ_PCINT2) return;
114 
115  auto& pcmsk = _pcmsk<Button>();
116  if (!(pcmsk & Pin::bit)) return; // irq for this bit is disabled
117 
118  const uint8_t p = GET_BIT(Pin, PIN);
119  const uint8_t pin_now = Button::Type == _buttons::Pressed_High ? p : !p;
120  uint8_t& pin_state = get_state<Button>();
121 
122  if ((pin_now && (pin_state & 1)) || (!pin_now && !(pin_state & 1))) return; // state hasn't changed
123 
124  _buttons::State new_state = (!pin_now) ? _buttons::Released : _buttons::Pressed;
125 
126  // don't act on any more irqs for some time (see run())
127  pcmsk &= ~_BV(Pin::bit);
128

232  uint8_t& state = get_state<Row, Column>();
233 
234  // check for new state
235  const uint8_t pin_now = !(GET_BIT(Column, PIN));
236 
237  if ((state / 2) < _buttons::block_rounds) {
238  state += 2;
239  return;
240  }

242  const uint8_t pin_changed = (pin_now && !(state & 1)) || (!pin_now && (state & 1));

245  state += 2; // keep bit0 untouched!
