743  /*
744  * There are 3 cases when we have reached the clock:
745  * (1) --p--t--c-- (i.e. prev smaller than target, target smaller than clock)
746  * current clock is higher than target. As prev is smaller than target, we
747  * passed the target.
748  *
749  * (2) -t-c-----p-
750  * Same analysis as prev. case. Except that we have wrapped.
751  *
752  * (3) -c-----p-t-
753  * In this case only clock has wrapped.
754  */
755  template<typename T>
756  static inline uint8_t clock_reached(const T& clock, const T& previous_clock, const T& target_clock) {
757  uint8_t ret; // this variable makes debugging easier
758  if (target_clock <= clock) {
759  ret = previous_clock <= target_clock || clock < previous_clock; // (1) & (2)
760  } else {
761  ret = previous_clock <= target_clock && previous_clock > clock; // (3)
762  }
763  return ret;
764  }
