241  constexpr static uint8_t _calc_prescaler() {
242  // is F_CPU divided by n (prescaler) below 200kHz? then prescaler is ok
243  // returns bits for ADPS2, ADPS1, ADPS0
244  return (F_CPU / 2) <= 200000 ? 0b000 :
245  (F_CPU / 4) <= 200000 ? 0b010 :
246  (F_CPU / 8) <= 200000 ? 0b011 :
247  (F_CPU / 16) <= 200000 ? 0b100 :
248  (F_CPU / 32) <= 200000 ? 0b101 :
249  (F_CPU / 64) <= 200000 ? 0b110 :
250  (F_CPU / 128) <= 200000 ? 0b111 :
251  0b000; // if nothing works only divide by 2 (safest prescaler)
252  }

301  PRR &= ~(_BV(PRADC)); // disable Power Reduction ADC
302 
303  const uint8_t ref = ((uint8_t) R) << REFS0;
304  const uint8_t source = I::bit << MUX0;
305  ADMUX = ref // set ref
306  | source; // set source
307  if (M != Mode::SingleConversion) {
308  ADCSRB |= (uint8_t) M << ADTS0;
309  ADCSRA |= _BV(ADATE);
310  }

313  | _calc_prescaler() << ADPS0; // set prescaler

318  if (!is_do_nothing_task) ADCSRA |= _BV(ADIE);

499  void _do_adc() {
500  irq_handler_for_adc_must_be_registered_for_noise_reduction();
501  uint8_t old_adie = ADCSRA & _BV(ADIE);
502 
503  if (!old_adie) ADCSRA |= _BV(ADIE);
504 
505  set_sleep_mode(SLEEP_MODE_IDLE);
506 
507  // Enable irq
508  NONATOMIC_BLOCK(NONATOMIC_RESTORESTATE) {
509  // By calling sleep_mode avr goes to sleep and automatically starts
510  // conversion.
511  sleep_mode();
512 
513  // We might wake up because of another IRQ. In that case simple wait
514  // until adc has finished.
515  busy_wait_adc_finished();
516  }
517 
518  // Restore ADIE if it wasn't enabled. Wait until adc has finished.
519  // Otherwise IRQ might not be called if CPU wakes up because of another
520  // IRQ.
521  if (!old_adie) {
522  ADCSRA &= ~(_BV(ADIE));
523  }
524  }
