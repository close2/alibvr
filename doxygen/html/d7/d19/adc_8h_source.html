260  constexpr static uint8_t _calc_prescaler() {
261  // is F_CPU divided by n (prescaler) below 200kHz? then prescaler is ok
262  // returns bits for ADPS2, ADPS1, ADPS0
263  return (F_CPU / 2) <= 200000 ? 0b000 :
264  (F_CPU / 4) <= 200000 ? 0b010 :
265  (F_CPU / 8) <= 200000 ? 0b011 :
266  (F_CPU / 16) <= 200000 ? 0b100 :

268  (F_CPU / 64) <= 200000 ? 0b110 :
