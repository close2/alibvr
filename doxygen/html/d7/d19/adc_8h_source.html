225  constexpr static uint8_t _calc_prescaler() {
226  // is F_CPU divided by n (prescaler) below 200kHz? then prescaler is ok
227  // returns bits for ADPS2, ADPS1, ADPS0
228  return (F_CPU / 2) <= 200000 ? 0b000 :
229  (F_CPU / 4) <= 200000 ? 0b010 :
230  (F_CPU / 8) <= 200000 ? 0b011 :
231  (F_CPU / 16) <= 200000 ? 0b100 :
232  (F_CPU / 32) <= 200000 ? 0b101 :
233  (F_CPU / 64) <= 200000 ? 0b110 :
234  (F_CPU / 128) <= 200000 ? 0b111 :
235  0b000; // if nothing works only divide by 2 (safest prescaler)
236  }

467  void _do_adc() {
468  irq_handler_for_adc_must_be_registered_for_noise_reduction();
469  uint8_t old_adie = ADCSRA & _BV(ADIE);
470 
471  if (!old_adie) ADCSRA |= _BV(ADIE);
472 
473  set_sleep_mode(SLEEP_MODE_IDLE);
474 
475  // Enable irq
476  NONATOMIC_BLOCK(NONATOMIC_RESTORESTATE) {
477  // By calling sleep_mode avr goes to sleep and automatically starts
478  // conversion.
479  sleep_mode();
480 
481  // We might wake up because of another IRQ. In that case simple wait
482  // until adc has finished.
483  busy_wait_adc_finished();
484  }
485 
486  // Restore ADIE if it wasn't enabled. Wait until adc has finished.
487  // Otherwise IRQ might not be called if CPU wakes up because of another
488  // IRQ.
489  if (!old_adie) {
490  ADCSRA &= ~(_BV(ADIE));
491  }
492  }
