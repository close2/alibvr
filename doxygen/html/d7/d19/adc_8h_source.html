101  constexpr static uint8_t _calc_prescaler() {
102  // is F_CPU divided by n (prescaler) below 200kHz? then prescaler is ok
103  // returns bits for ADPS2, ADPS1, ADPS0
104  return (F_CPU / 2) <= 200000 ? 0b000 :
105  (F_CPU / 4) <= 200000 ? 0b010 :
106  (F_CPU / 8) <= 200000 ? 0b011 :
107  (F_CPU / 16) <= 200000 ? 0b100 :
108  (F_CPU / 32) <= 200000 ? 0b101 :
109  (F_CPU / 64) <= 200000 ? 0b110 :
110  (F_CPU / 128) <= 200000 ? 0b111 :
111  0b000; // if nothing works only divide by 2 (safest prescaler)
112  }

137  const uint8_t ref = ((uint8_t) Ref) << REFS0;

234  void _do_adc() {
235  irq_handler_for_adc_must_be_registered_for_noise_reduction();
236  uint8_t old_adie = ADCSRA & _BV(ADIE);
237 
238  if (!old_adie) ADCSRA |= _BV(ADIE);
239 
240  set_sleep_mode(SLEEP_MODE_IDLE);
241 
242  // Enable irq
243  NONATOMIC_BLOCK(NONATOMIC_RESTORESTATE) {
244  // By calling sleep_mode avr goes to sleep and automatically starts
245  // conversion.
246  sleep_mode();
247 
248  // We might wake up because of another IRQ. In that case simple wait
249  // until adc has finished.
250  busy_wait_adc_finished();
251  }
252 
253  // Restore ADIE if it wasn't enabled. Wait until adc has finished.
254  // Otherwise IRQ might not be called if CPU wakes up because of another
255  // IRQ.
256  if (!old_adie) {
257  ADCSRA &= ~(_BV(ADIE));
258  }
259  }
