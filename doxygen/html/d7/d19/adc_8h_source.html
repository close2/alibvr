233  // is F_CPU divided by n (prescaler) below 200kHz? then prescaler is ok
234  // returns bits for ADPS2, ADPS1, ADPS0
235  return (F_CPU / 2) <= 200000 ? 0b000 :

474  void _do_adc() {
475  irq_handler_for_adc_must_be_registered_for_noise_reduction();
476  uint8_t old_adie = ADCSRA & _BV(ADIE);
477 
478  if (!old_adie) ADCSRA |= _BV(ADIE);
479 
480  set_sleep_mode(SLEEP_MODE_IDLE);
481 
482  // Enable irq
483  NONATOMIC_BLOCK(NONATOMIC_RESTORESTATE) {
484  // By calling sleep_mode avr goes to sleep and automatically starts
485  // conversion.
486  sleep_mode();
487 
488  // We might wake up because of another IRQ. In that case simple wait
489  // until adc has finished.
490  busy_wait_adc_finished();
491  }
492 
493  // Restore ADIE if it wasn't enabled. Wait until adc has finished.
494  // Otherwise IRQ might not be called if CPU wakes up because of another
495  // IRQ.
496  if (!old_adie) {
497  ADCSRA &= ~(_BV(ADIE));
498  }
499  }
