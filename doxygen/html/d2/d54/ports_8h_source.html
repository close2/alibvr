487  template <enum DataDirection Dd, class Io>
488  uint8_t _set_or_get(uint8_t val) {
489  static_assert(Io::io_reg == _IOReg::DDRx ||
490  Io::io_reg == _IOReg::PINx ||
491  Io::io_reg == _IOReg::PORTx,
492  "IO Register is unknown. Only Ddr, Pin and Port are supported.");
493  switch(Io::io_reg) {
494  case _IOReg::DDRx:
495  switch(Io::port) {
496  case _Port::B:
497  return _set_or_get_f<Dd>(DDRB, Io::bit, val);
498  case _Port::C:
499  return _set_or_get_f<Dd>(DDRC, Io::bit, val);
500  case _Port::D:
501  return _set_or_get_f<Dd>(DDRD, Io::bit, val);
502  }
503  break;
504  case _IOReg::PORTx:
505  switch(Io::port) {
506  case _Port::B:
507  return _set_or_get_f<Dd>(PORTB, Io::bit, val);
508  case _Port::C:
509  return _set_or_get_f<Dd>(PORTC, Io::bit, val);
510  case _Port::D:
511  return _set_or_get_f<Dd>(PORTD, Io::bit, val);
512  }
513  break;
514  case _IOReg::PINx:
515  switch(Io::port) {
516  case _Port::B:
517  return _set_or_get_f<Dd>(PINB, Io::bit, val);
518  case _Port::C:
519  return _set_or_get_f<Dd>(PINC, Io::bit, val);
520  case _Port::D:
521  return _set_or_get_f<Dd>(PIND, Io::bit, val);
522  }
523  break;
524  }
525  return 0; // should never happen
526  }

558  const V& val) {
559  uint8_t vB = 0;
560  uint8_t vC = 0;
561  uint8_t vD = 0;
562  uint8_t maskB = 0;
563  uint8_t maskC = 0;
564  uint8_t maskD = 0;

566  const uint8_t d0isB = D0::port == NS::_Port::B; const uint8_t d0isC = D0::port == NS::_Port::C;
567  const uint8_t d1isB = D1::port == NS::_Port::B; const uint8_t d1isC = D1::port == NS::_Port::C;
568  const uint8_t d2isB = D2::port == NS::_Port::B; const uint8_t d2isC = D2::port == NS::_Port::C;
569  const uint8_t d3isB = D3::port == NS::_Port::B; const uint8_t d3isC = D3::port == NS::_Port::C;

571  const uint8_t d5isB = D5::port == NS::_Port::B; const uint8_t d5isC = D5::port == NS::_Port::C;
572  const uint8_t d6isB = D6::port == NS::_Port::B; const uint8_t d6isC = D6::port == NS::_Port::C;
573  const uint8_t d7isB = D7::port == NS::_Port::B; const uint8_t d7isC = D7::port == NS::_Port::C;
574 
575  uint8_t& v0 = d0isB ? vB : (d0isC ? vC : vD);
576  uint8_t& v1 = d1isB ? vB : (d1isC ? vC : vD);
577  uint8_t& v2 = d2isB ? vB : (d2isC ? vC : vD);
578  uint8_t& v3 = d3isB ? vB : (d3isC ? vC : vD);
579  uint8_t& v4 = d4isB ? vB : (d4isC ? vC : vD);
580  uint8_t& v5 = d5isB ? vB : (d5isC ? vC : vD);
581  uint8_t& v6 = d6isB ? vB : (d6isC ? vC : vD);
582  uint8_t& v7 = d7isB ? vB : (d7isC ? vC : vD);
583 
584  uint8_t& mask0 = d0isB ? maskB : (d0isC ? maskC : maskD);
585  uint8_t& mask1 = d1isB ? maskB : (d1isC ? maskC : maskD);
586  uint8_t& mask2 = d2isB ? maskB : (d2isC ? maskC : maskD);
587  uint8_t& mask3 = d3isB ? maskB : (d3isC ? maskC : maskD);
588  uint8_t& mask4 = d4isB ? maskB : (d4isC ? maskC : maskD);
589  uint8_t& mask5 = d5isB ? maskB : (d5isC ? maskC : maskD);
590  uint8_t& mask6 = d6isB ? maskB : (d6isC ? maskC : maskD);
591  uint8_t& mask7 = d7isB ? maskB : (d7isC ? maskC : maskD);
592 
593  // avoid warning: left shift count >= width of type
594  // even if not used. (if compiled without optimization)
595  if (D0::bit < 8 && B0 < 8) {
596  const uint8_t shift0 = (D0::bit >= 8) ? 0 : D0::bit;
597  mask0 |= 1 << shift0;
598  if (Dd == NS::DataDirection::Write) v0 |= val & _BV(B0);
599  }
600  if (D1::bit < 8 && B1 < 8) {
601  const uint8_t shift1 = (D1::bit >= 8) ? 0 : D1::bit;
602  mask1 |= 1 << shift1;
603  if (Dd == NS::DataDirection::Write) v1 |= val & _BV(B1);
604  }
605  if (D2::bit < 8 && B2 < 8) {
606  const uint8_t shift2 = (D2::bit >= 8) ? 0 : D2::bit;
607  mask2 |= 1 << shift2;
608  if (Dd == NS::DataDirection::Write) v2 |= val & _BV(B2);
609  }
610  if (D3::bit < 8 && B3 < 8) {
611  const uint8_t shift3 = (D3::bit >= 8) ? 0 : D3::bit;
612  mask3 |= 1 << shift3;
613  if (Dd == NS::DataDirection::Write) v3 |= val & _BV(B3);
614  }
615  if (D4::bit < 8 && B4 < 8) {
616  const uint8_t shift4 = (D4::bit >= 8) ? 0 : D4::bit;
617  mask4 |= 1 << shift4;
618  if (Dd == NS::DataDirection::Write) v4 |= val & _BV(B4);
619  }
620  if (D5::bit < 8 && B5 < 8) {
621  const uint8_t shift5 = (D5::bit >= 8) ? 0 : D5::bit;
622  mask5 |= 1 << shift5;
623  if (Dd == NS::DataDirection::Write) v5 |= val & _BV(B5);
624  }
625  if (D6::bit < 8 && B6 < 8) {
626  const uint8_t shift6 = (D6::bit >= 8) ? 0 : D6::bit;
627  mask6 |= 1 << shift6;
628  if (Dd == NS::DataDirection::Write) v6 |= val & _BV(B6);

630  if (D7::bit < 8 && B7 < 8) {
631  const uint8_t shift7 = (D7::bit >= 8) ? 0 : D7::bit;
632  mask7 |= 1 << shift7;
633  if (Dd == NS::DataDirection::Write) v7 |= val & _BV(B7);
634  }
635 
636  if (maskB != 0) {
637  if (Dd == NS::DataDirection::Write) rB = (rB & ~maskB) | vB;
638  else vB = rB;
639  }
640  if (maskC != 0) {

642  else vC = rC;

644  if (maskD != 0) {
645  if (Dd == NS::DataDirection::Write) rD = (rD & ~maskD) | vD;
646  else vD = rD;
647  }
648 
649  uint8_t ret = 0;
650  if (Dd == NS::DataDirection::Read) {
651  ret |= (v0 & _BV(D0::bit)) ? _BV(B0) : 0;
652  ret |= (v1 & _BV(D1::bit)) ? _BV(B1) : 0;
653  ret |= (v2 & _BV(D2::bit)) ? _BV(B2) : 0;
654  ret |= (v3 & _BV(D3::bit)) ? _BV(B3) : 0;
655  ret |= (v4 & _BV(D4::bit)) ? _BV(B4) : 0;
656  ret |= (v5 & _BV(D5::bit)) ? _BV(B5) : 0;
657  ret |= (v6 & _BV(D6::bit)) ? _BV(B6) : 0;
658  ret |= (v7 & _BV(D7::bit)) ? _BV(B7) : 0;
659  }
660  return ret;
661  }
