485  template <enum DataDirection Dd, class Io>
486  uint8_t _set_or_get(uint8_t val) {
487  static_assert(Io::io_reg == IOReg::DDRx ||
488  Io::io_reg == IOReg::PINx ||
489  Io::io_reg == IOReg::PORTx,
490  "IO Register is unknown. Only Ddr, Pin and Port are supported.");
491  switch(Io::io_reg) {
492  case IOReg::DDRx:
493  switch(Io::port) {
494  case _Port::B:
495  return _set_or_get_f<Dd>(DDRB, Io::bit, val);
496  case _Port::C:
497  return _set_or_get_f<Dd>(DDRC, Io::bit, val);
498  case _Port::D:
499  return _set_or_get_f<Dd>(DDRD, Io::bit, val);
500  }
501  break;
502  case IOReg::PORTx:
503  switch(Io::port) {
504  case _Port::B:
505  return _set_or_get_f<Dd>(PORTB, Io::bit, val);
506  case _Port::C:
507  return _set_or_get_f<Dd>(PORTC, Io::bit, val);
508  case _Port::D:
509  return _set_or_get_f<Dd>(PORTD, Io::bit, val);
510  }
511  break;
512  case IOReg::PINx:
513  switch(Io::port) {
514  case _Port::B:
515  return _set_or_get_f<Dd>(PINB, Io::bit, val);
516  case _Port::C:
517  return _set_or_get_f<Dd>(PINC, Io::bit, val);
518  case _Port::D:

520  }
