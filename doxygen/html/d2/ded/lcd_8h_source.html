49  // If d7 - d4 is alligned on PORTx_3 - PORTx_0 and reg_select is on PORTx as
50  // well, shift the reg select bit, so that a single assignment can be used.
51  // Returns the bit position of reg_select on our data byte.
52  template <typename LCD_D7, typename LCD_D6, typename LCD_D5, typename LCD_D4, typename LCD_ENABLE, typename LCD_REG_SELECT>
53  constexpr inline uint8_t _reg_select_bit() {
54  const uint8_t reg_sel = LCD_REG_SELECT::bit;
55  if (reg_sel > 3) return reg_sel;
56  else return _lcd::default_reg_select_bit;
57  }

62  template <typename LCD_D7, typename LCD_D6, typename LCD_D5, typename LCD_D4, typename LCD_ENABLE, typename LCD_REG_SELECT, enum _lcd::Command_Data Command_Or_Data>
63  static inline void _lcd_send_nibble(const uint8_t& data_4_bit) {
64  // reg_select might be on another port in which case calculating the
65  // position of the reg_bit doesn't help. If however it is on the same port
66  // and the data bits are aligned, assigning the data bits and the reg bit
67  // might be possible in one PORT assignment.
68  const uint8_t reg_bit = _reg_select_bit<LCD_D7, LCD_D6, LCD_D5, LCD_D4, LCD_ENABLE, LCD_REG_SELECT>();
69  const uint8_t data = data_4_bit | (Command_Or_Data << reg_bit);
70  if (reg_bit == 4) set_8_bits<LCD_D7, 3, LCD_D6, 2, LCD_D5, 1, LCD_D4, 0, LCD_REG_SELECT, 4, PIN_UNUSED, -1, PIN_UNUSED, -1, PIN_UNUSED, -1>(PORTB, PORTC, PORTD, data);
71  if (reg_bit == 5) set_8_bits<LCD_D7, 3, LCD_D6, 2, LCD_D5, 1, LCD_D4, 0, LCD_REG_SELECT, 5, PIN_UNUSED, -1, PIN_UNUSED, -1, PIN_UNUSED, -1>(PORTB, PORTC, PORTD, data);
72  if (reg_bit == 6) set_8_bits<LCD_D7, 3, LCD_D6, 2, LCD_D5, 1, LCD_D4, 0, LCD_REG_SELECT, 6, PIN_UNUSED, -1, PIN_UNUSED, -1, PIN_UNUSED, -1>(PORTB, PORTC, PORTD, data);
73  if (reg_bit == 7) set_8_bits<LCD_D7, 3, LCD_D6, 2, LCD_D5, 1, LCD_D4, 0, LCD_REG_SELECT, 7, PIN_UNUSED, -1, PIN_UNUSED, -1, PIN_UNUSED, -1>(PORTB, PORTC, PORTD, data);
74  _delay_us(1); // tAS Address Setup Time (140ns min) (tAS time of REG_SELECT before ENABLE becomes 1)
75  // note that we could set DATA together with ENABLE, but setting it with data_select is fine
76  SET_BIT(LCD_ENABLE, PORT, 1);
77  _delay_us(1); // tEH Enable High Time (450ns min) and tDS Data Set up time (200ns min) (tDS == time DATA is valid till ENABLE becomes 0)
78  // tDA Data Access time (320ns max) is < 0ns because we set DATA together with REG_SELECT (tDA == time between ENABLE 1 and DATA valid)
79  SET_BIT(LCD_ENABLE, PORT, 0);
80  _delay_us(1); // tAH, tDH and tEL just to be sure
81  // tAH Address hold time (10ns min) (tAH == time SELECT is still valid after ENABLE becomes 0)
82  // tDH Address hold time (20ns min) (tDH == time DATA is still valid after ENABLE becomes 0)
83  // tEL Enable low time (500ns min)
84  }
