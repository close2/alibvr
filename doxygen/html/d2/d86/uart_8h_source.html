209  // implement IRQ-handler interface
210  // this handle function handles 2 irqs USART_RX and USART_UDRE!
211  template <class I>
212  static inline void handle(I) {
213  if (!use_irqs) return;
214 
215  if (I::irq == _irqs::IRQ_USART_RX) {
216  // we need to first store the UCSR0A before reading from UDR0 because
217  // the receiver could possibly overwrite UCSR0A with the status for
218  // the next byte (which will be received as soon as we read from UDR0)
219  const uint8_t status = UCSR0A & (_BV(FE0) | _BV(DOR0) | _BV(UPE0));
220  const uint8_t c = UDR0;
221  if (status == 0) {
222  if (rx_buffer_size > 0) {
223  rx_buffer::push(c);
224  }
225  IrqTask::rx(c);
226  }
227  } else {
228  // _irqs::IRQ_USART_UDRE
229  uint8_t c;
230  if (!IrqTask::tx_is_empty()) {
231  c = IrqTask::tx_pop();
232  } else {
233  if (tx_buffer_size == 0 || tx_buffer::is_empty()) {
234  UCSR0B &= ~_BV(UDRIE0);
235  IrqTask::tx_done();
236  return;
237  }
238  c = tx_buffer::pop();
239  }
240 
241  UDR0 = c;
242  }
