24 /*
25  * Difference between coils and discrete inputs:
26  *
27  * copied from: http://www.ni.com/white-paper/7675/en/
28  *
29  * The Discrete Inputs represent a single bit (Boolean) which can only be read.
30  * In other words, the master can only perform a read action on the discrete inputs.
31  * The same holds for the Input Registers.
32  * The master can only read the slave’s Input Registers.
33  * The difference between the Discrete Inputs and the Input Registers is that
34  * the Input Registers represent 16 bits while the Discrete Inputs are only a
35  * single bit.
36  * The Coils also represent a Boolean data type which can be read and written to by the master.
37  * The Holding Registers represent a 16 bit word that can be read and written to.
38  *
39  * +-------+----------------+-------------+
40  * | | read | r/w |
41  * +-------+----------------+-------------+
42  * | 1bit | discrete input | coil |
43  * | 16bit | input reg | holding reg |
44  * +-------+----------------+-------------+
45  *
46  */

236  template <typename T>
237  static T run(const T& clock) {
238  const uint8_t pos = get_pos();
239  uint8_t* message = get_message();
240 
241  // for now allow messages > than 6 even though we don't support them
242  if (pos < 6) {
243  _error();
244  }
245 
246  // message[0] is our addressId or 0
247  const uint8_t functionCode = message[1];
248 
249  const uint16_t address = message[2] << 8 | message[3];
250  _modbus::ReadWrite readWrite;
251  _modbus::DataType type;
252 
253  // value or count of read values
254  const uint16_t value = message[4] << 8 | message[5];
255 
256  uint8_t bitOperation;
257 
258  switch(functionCode) {
259  case _modbus::ReadDiscreteInputs:
260  readWrite = _modbus::Read;
261  type = _modbus::DiscreteInput;
262  bitOperation = 1;
263  break;
264  case _modbus::ReadCoils:
265  readWrite = _modbus::Read;
266  type = _modbus::Coil;
267  bitOperation = 1;
268  break;
269  case _modbus::WriteSingleCoil:
270  readWrite = _modbus::Write;
271  type = _modbus::Coil;
272  bitOperation = 1;
273  break;
274  case _modbus::ReadInputRegisters:
275  readWrite = _modbus::Read;
276  type = _modbus::InputRegister;
277  bitOperation = 0;
278  break;
279  case _modbus::ReadHoldingRegisters:
280  readWrite = _modbus::Read;
281  type = _modbus::Register;
282  bitOperation = 0;
283  break;
284  case _modbus::WriteSingleRegister:
285  readWrite = _modbus::Write;
286  type = _modbus::Register;
287  bitOperation = 0;
288  break;
289  default:
290  _exception(_modbus::IllegalFunction);
291  return 0;
292  }
293 
294  // we do not support multiple read operations
295  if (readWrite == _modbus::Read && value != 1) {
296  _exception(_modbus::IllegalDataAddress);
297  return 0;
298  }
299 
300  // only message with length 6 are possible, otherwise we received an invalid frame
301  if (pos != 6) {
302  _error();
303  }
304 
305  if (readWrite == _modbus::Write) {
306  uint16_t v = value;
307  if (!_modbus::execModbusData<_modbus::Write>(v, type, address, ModbusDataList())) {
308  _exception(_modbus::IllegalDataAddress);
309  return 0;
310  }
311  if (message[0] == 0) {
312  // broadcast
313  _restart();
314  return 0;
315  }
316  // echo the message back to sender
317  message[0] = pos;
318  } else {
319  uint16_t v = 0;
320  if (!_modbus::execModbusData<_modbus::Read>(v, type, address, ModbusDataList())) {
321  _exception(_modbus::IllegalDataAddress);
322  return 0;
323  }
324 
325  if (message[0] == 0) {
326  // broadcast
327  _restart();
328  return 0;
329  }
330  if (bitOperation) {
331  v = (v != 0) ? 1 : 0;
332  message[2] = 1;
333  message[3] = v;
334  message[0] = 4;
335  } else {
336  message[2] = 2;
337  message[3] = v >> 8;
338  message[4] = v & 0x00FF;
339  message[0] = 5;
340  }
341  }
342  _send_reply();
343 
344  // we wan't to be called right away if we are enabled. (which we are, when a
345  // complete message has been received)
346  return 0;
347  }
